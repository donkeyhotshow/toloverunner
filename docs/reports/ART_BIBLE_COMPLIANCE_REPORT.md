**Детали реализации:**
```typescript
// WorldBendingShader.ts
export function applyWorldBending(material, options = {}) {
    const { curvature = 0.005, enabled = true } = options;
    // Патчим vertex shader
    worldPos.y -= distFromCamera * distFromCamera * curvatureFactor;
}

// BioTrackSegment.tsx
useEffect(() => {
    applyWorldBending(trackMaterial, { curvature: 0.003, enabled: true });
    applyWorldBending(borderMaterial, { curvature: 0.003, enabled: true });
    applyWorldBending(cellMaterial, { curvature: 0.003, enabled: true });
}, []);
```

**Эффект:** ✅ Мир загибается вниз, спавн объектов скрыт, ощущение бесконечности
